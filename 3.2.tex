  (define (make-monitored f)
    (let ((count 0))
      (lambda (m)
        (cond ((eq? m 'how-many-call?) count)
              ((eq? m 'reset-count) (set! count 0))
              (else
               (set! count (+ count 1))
               (f m))))))
\end{lstlisting}
