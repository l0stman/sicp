(define (c* x y)
  (let ((z (make-connector)))
    (multiplier x y z)
    z))
(define (c/ x y)
  (let ((z (make-connector)))
    (multiplier y z x)
    z))
(define (cv v)
  (let ((x (make-connectr)))
    (constant v x)
    x))
\end{lstlisting}
