If we add the following piece of code in the polynomial package:
\begin{lstlisting}
(define (=zero? poly)
  (define (iter t)
    (cond ((empty-termlist? t) true)
          ((zero? (coeff (first-term t)))
           (iter (rest-terms t)))
          (else false)))
  (iter (term-list poly)))
(put '=zero? '(polynomial) =zero?)
\end{lstlisting}
