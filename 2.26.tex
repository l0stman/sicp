\documentclass[a4paper,12pt]{article}
\usepackage{listings}
\lstset{language=Lisp}

\begin{document}
\begin{lstlisting}
  (define x (list 1 2 3))
  (define y (list 4 5 6))
  (append x y)
\end{lstlisting}
$\rightarrow$ \lstinline!(1 2 3 4 5 6)!
\begin{lstlisting}
  (cons x y)
\end{lstlisting}
$\rightarrow$ \lstinline!((1 2 3) 4 5 6)!
\begin{lstlisting}
  (list x y)
\end{lstlisting}
$\rightarrow$ \lstinline!((1 2 3) (4 5 6))!
\end{document}
