(define (unique-triple n)
  (flatmap (lambda (i)
             (flatmap (lambda (j)
                    (map (lambda (k)
                           (list i j k))
                         (enumerate-interval 1 (- j 1))))
                  (enumerate-interval 1 (- i 1))))
           (enumerate-interval 1 n)))
(define (triple-sum s n)
  (filter (lambda (t)
            (= s (foldr + 0 t)))
          (unique-triple n)))
