(define (rlc r l c dt)
  (lambda (vc0 il0)
    (define vc (integral (delay (scale-stream il (/ -1 c)))
                         vc0
                         dt))
    (define il (integral (delay
                           (add-streams
                            (scale-stream vc (/ 1 l))
                            (scale-stream il (/ (- r) l))))
                         il0
                         dt))
    (stream-map cons vc il)))
