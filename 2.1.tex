(define (make-rat n d)
  (define (sign x)
    (if (< x 0) -1 1))
  (let ((g (gcd n d)))
    (cons (* (sign n)
             (sign d)
             (/ (abs n) g))
          (/ (abs d) g))))
\end{lstlisting}
