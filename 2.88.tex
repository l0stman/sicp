We add the following definitions in the polynomial package
\begin{lstlisting}
(define (sub-poly p1 p2)
  (add-poly p1
            (make-poly
             (variable p2)
             (mul-terms (term-list p2)
                        (adjoin-term (make-term 0 -1)
                                     (the-empty-termlist))))))
(put 'sub '(polynomial) sub-poly)
\end{lstlisting}
